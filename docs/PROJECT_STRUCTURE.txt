#### 工具调用逻辑（Line 258-367）：

```typescript
apiRoutes.post('/sessions/:sessionId/chat', async (c) => {
  // 1. 保存访客消息
  // 2. 检查主人是否在线
  // 3. 构建上下文（最近 40 条消息）
  
  // 4. 第一次调用 OpenAI（传入工具列表）
  let resp = await fetch(apiUrl, {
    body: JSON.stringify({
      model: 'MiniMax-M2.5',
      messages,
      tools,  // ← 传入工具定义
      tool_choice: 'auto'
    })
  })
  
  // 5. 检查是否需要调用工具
  if (message?.tool_calls) {
    // 6. 执行工具
    for (const toolCall of message.tool_calls) {
      const result = await executeToolCall(...)
      toolMessages.push({ role: 'tool', content: result })
    }
    
    // 7. 第二次调用 OpenAI（传入工具结果）
    resp = await fetch(apiUrl, {
      body: JSON.stringify({
        messages: [...messages, ...toolMessages]
      })
    })
  }
  
  // 8. 保存 AI 回复并返回
  return c.json({ aiReply: { ... } })
})
```

### 3. `src/routes/visitor.ts`

**访客页面路由**

```typescript
export const visitorRoutes = new Hono()

// 主页
visitorRoutes.get('/', (c) => {
  return c.html(`
    <h1>欢迎访问 my-agent</h1>
    <a href="/u/jennia">示例：与 jennia 的 AI 助手对话</a>
  `)
})

// 访客对话页面
visitorRoutes.get('/u/:ownerId', async (c) => {
  const owner = await getOwner(c.env.KV, ownerId)
  if (!owner) return c.html('主人不存在', 404)
  
  return c.html(visitorChatPage(owner))  // 返回聊天页面 HTML
})

// visitorChatPage 函数生成完整 HTML
function visitorChatPage(owner: any): string {
  return `<!DOCTYPE html>
    <html>
      <head>
        <title>和 ${owner.ai_name} 聊天 · ${owner.name}</title>
        <script src="/static/visitor-chat.js"></script>
      </head>
      <body>...</body>
    </html>
  `
}
```

## 🚀 开发流程

### 本地开发

```bash
# 1. 安装依赖
npm install

# 2. 配置环境变量（.dev.vars）
OPENAI_API_KEY=your-key
OPENAI_BASE_URL=https://llm.mcisaas.com

# 3. 构建项目
npm run build

# 4. 启动开发服务器
npm run dev:sandbox

# 5. 访问
http://localhost:3000/u/jennia
```

### 添加新工具

```bash
# 1. 编辑 src/routes/api.ts
#    - 在 tools 数组添加定义
#    - 在 executeToolCall 添加实现

# 2. 构建并部署
npm run build
npm run deploy

# 3. 测试
bash scripts/test-tools.sh
```
